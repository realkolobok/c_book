#include <stdio.h>

void in_comment();
void in_string();


int main()
{
    int c;
    while ((c = getchar()) != EOF) {
        if (c == '/') {
            if ((c = getchar()) != '*') {
                putchar('/');
                putchar(c);
            } else {
                in_comment();
            }
        } else if (c == '') {
            in_string();
        } else {
            putchar(c);
        }
    }
    return 0;
}

void in_comment()
{
    int last_c, c;
    last_c = getchar();
    while ((c = getchar()) != EOF) {
        if (last_c == '*' && c == '/') {
            return;
        }
        last_c = c;
    }
}

void in_string()
{
    int c;
    while ((c = getchar()) != EOF) {
        putchar(c);
        if (c == '"') {
            return;
        }
    }
}
